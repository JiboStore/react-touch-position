import React from 'react';
import ReactTouchPosition from 'react-touch-position';

import TouchPositionLabel from './TouchPositionLabel';
import onOnPositionChanged from './onOnPositionChanged';
import InstructionsLabel from './InstructionsLabel';

export default class extends React.Component {
    constructor(props) {
        super(props);
        this.state = {
            isPositionOutside: true,
            touchPosition: {
                x: 0,
                y: 0,
            }
        }
    }

    render() {
        return (
            <div className="example-container">
                <ReactTouchPosition  {...{
                    className: 'example',
                    onOnPositionChanged: ({ isPositionOutside, touchPosition }) => {
                        this.setState({ isPositionOutside, touchPosition });
                    },
                    shouldDecorateChildren: false
                }}>
                    <TouchPositionLabel />
                    <InstructionsLabel />
                </ReactTouchPosition>
                <OnPositionChangeLabel {...this.state} />
            </div>
        );
    }
}
